Example
function MyComponent() {
  const { uploadedFile, fetchUploadedFile, handleFileChange } = useFileOperations("example-file");

  useEffect(() => {
    fetchUploadedFile();
  }, []);

  return (
    <div>
      {uploadedFile && <img src={uploadedFile} alt="Uploaded file" />}
      <input type="file" onChange={handleFileChange} />
    </div>
  );
}
